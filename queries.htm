  SELECT 
    t.name, 
    s.script_id, 
    s.date, 
    count(st.script_id) `number of successors`, 
    t.is_interactive 
  FROM 
    tutorials t 
    join `scripts_tutorials` st on st.tutorial_id = t.id 
    join scripts s on s.id = t.script_id 
  where 
    t.is_interactive = true 
  group by 
    s.id 
  order by 
    N desc
) 
union all
